154{

156 const Volume n1 = acGetMaxNN(info);
157 const Volume m0 = (Volume){0, 0, 0};
158 const Volume m1 = acGetLocalMM(info);

162 as_size_t(info.int3_params[AC_reduction_tile_dimensions].x),
163 as_size_t(info.int3_params[AC_reduction_tile_dimensions].y),
164 as_size_t(info.int3_params[AC_reduction_tile_dimensions].z)
165 };
166
167 return (AcMeshDims){
168 .n0 = n0,
169 .n1 = n1,
170 .m0 = m0,
171 .m1 = m1,
172 .nn = nn,
173 .reduction_tile = reduction_tile,
174 };
175}
