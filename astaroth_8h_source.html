415static inline AcMeshDims
416acGetMeshDims(const AcMeshInfo info, const VertexBufferHandle vtxbuf)
417{
418 const int3 halos = acGetFieldHalos(info,vtxbuf);
419 const Volume n0 =
420 (Volume)
421 {
422 as_size_t(halos.x),
423 as_size_t(halos.y),
424 as_size_t(halos.z)
425 };
426 const Volume m1 =
427 (Volume){
428 as_size_t(info.int3_params[vtxbuf_dims[vtxbuf]].x),
429 as_size_t(info.int3_params[vtxbuf_dims[vtxbuf]].y),
430 as_size_t(info.int3_params[vtxbuf_dims[vtxbuf]].z)
431 };
432 const Volume n1 =
433 (Volume)
434 {
435 m1.x-n0.x,
436 m1.y-n0.y,
437 m1.z-n0.z,
438 };
439 const Volume m0 = (Volume){0, 0, 0};
440 const Volume nn =
441 (Volume)
442 {
443 m1.x-2*n0.x,
444 m1.y-2*n0.y,
445 m1.z-2*n0.z,
446 };
447 const Volume reduction_tile = (Volume)
448 {
449 as_size_t(info.int3_params[AC_reduction_tile_dimensions].x),
450 as_size_t(info.int3_params[AC_reduction_tile_dimensions].y),
451 as_size_t(info.int3_params[AC_reduction_tile_dimensions].z)
452 };
453
454 return (AcMeshDims){
455 .n0 = n0,
456 .n1 = n1,
457 .m0 = m0,
458 .m1 = m1,
459 .nn = nn,
460 .reduction_tile = reduction_tile,
461 };
462}

1012acUpdateDecompositionParams(AcMeshInfo* dst)
1013{
1014#if AC_MPI_ENABLED
1015 int nprocs{};
1016 int rank{};
1017 ERRCHK_ALWAYS(dst->comm != NULL && dst->comm->handle != MPI_COMM_NULL);
1018 MPI_Comm_size(dst->comm->handle,&nprocs);
