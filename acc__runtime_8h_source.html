597 template <typename P>
598 constexpr static auto
599 get_array_dim_sizes(const P array, const AcMeshInfo info)
600 {
601 auto dims = get_array_info(array).dims;
602 int num_dims = get_array_info(array).num_dims;
603 std::array<size_t,20> res{};
604 for(int i = 0; i < num_dims; ++i)
605 {
606 if(!dims[i].from_config)
607 {
608 res[i] = dims[i].base;
609 continue;
610 }
611 if(dims[i].member == NULL)
612 {
613 if(dims[i].run_const)
614 {
615 res[i] = info.run_consts.config.int_params[dims[i].base];
616 }
617 else
618 {
619 res[i] = info.int_params[dims[i].base];
620 }
621 continue;
622 }
623 if(!strcmp(dims[i].member,"x"))
624 {
625 res[i] = info.int3_params[dims[i].base].x;
626 continue;
627 }
628 if(!strcmp(dims[i].member,"y"))
629 {
630 res[i] = info.int3_params[dims[i].base].y;
631 continue;
632 }
633 if(!strcmp(dims[i].member,"z"))
634 {
635 res[i] = info.int3_params[dims[i].base].z;
636 continue;
637 }
638 }
639 return res;
640 }
