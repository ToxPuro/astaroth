297int hashmap_create(const hashmap_uint32_t initial_capacity,
298 struct hashmap_s *const out_hashmap) {
299 struct hashmap_create_options_s options;
300 memset(&options, 0, sizeof(options));
301 options.initial_capacity = initial_capacity;
302
303 return hashmap_create_ex(options, out_hashmap);
304}
305
306int hashmap_create_ex(struct hashmap_create_options_s options,
307 struct hashmap_s *const out_hashmap) {
308 if (2 > options.initial_capacity) {
309 options.initial_capacity = 2;
310 } else if (0 != (options.initial_capacity & (options.initial_capacity - 1))) {
311 options.initial_capacity = 1u
312 << (32 - hashmap_clz(options.initial_capacity));
313 }
314
315 if (HASHMAP_NULL == options.hasher) {
316 options.hasher = &hashmap_crc32_hasher;
317 }
318
319 if (HASHMAP_NULL == options.comparer) {
320 options.comparer = &hashmap_memcmp_comparer;
321 }
322
323 out_hashmap->data = HASHMAP_CAST(
324 struct hashmap_element_s *,
325 calloc(options.initial_capacity + HASHMAP_LINEAR_PROBE_LENGTH,
326 sizeof(struct hashmap_element_s)));
327
328 out_hashmap->log2_capacity = 31 - hashmap_clz(options.initial_capacity);
329 out_hashmap->size = 0;
330 out_hashmap->hasher = options.hasher;
331 out_hashmap->comparer = options.comparer;
332
333 return 0;
334}

336int hashmap_put(struct hashmap_s *const m, const void *const key,
337 const hashmap_uint32_t len, void *const value) {
338 hashmap_uint32_t index;
339
340 if ((HASHMAP_NULL == key) || (0 == len)) {
341 return 1;
342 }
343
344 /* Find a place to put our value. */
345 while (!hashmap_hash_helper(m, key, len, &index)) {
346 if (hashmap_rehash_helper(m)) {
347 return 1;
348 }
349 }
350
351 /* Set the data. */
352 m->data[index].data = value;
353 m->data[index].key = key;
354 m->data[index].key_len = len;
355
356 /* If the hashmap element was not already in use, set that it is being used
357 * and bump our size. */
358 if (0 == m->data[index].in_use) {
359 m->data[index].in_use = 1;
360 m->size++;
361 }
362
363 return 0;
364}

366void *hashmap_get(const struct hashmap_s *const m, const void *const key,
367 const hashmap_uint32_t len) {
368 hashmap_uint32_t i, curr;
369
370 if ((HASHMAP_NULL == key) || (0 == len)) {
371 return HASHMAP_NULL;
372 }
373
374 curr = hashmap_hash_helper_int_helper(m, key, len);
375
376 /* Linear probing, if necessary */
377 for (i = 0; i < HASHMAP_LINEAR_PROBE_LENGTH; i++) {
378 const hashmap_uint32_t index = curr + i;
379
380 if (m->data[index].in_use) {
381 if (m->comparer(m->data[index].key, m->data[index].key_len, key, len)) {
382 return m->data[index].data;
383 }
384 }
385 }
386
387 /* Not found */
388 return HASHMAP_NULL;
389}

391int hashmap_remove(struct hashmap_s *const m, const void *const key,
392 const hashmap_uint32_t len) {
393 hashmap_uint32_t i, curr;
394
395 if ((HASHMAP_NULL == key) || (0 == len)) {
396 return 1;
397 }
398
399 curr = hashmap_hash_helper_int_helper(m, key, len);
400
401 /* Linear probing, if necessary */
402 for (i = 0; i < HASHMAP_LINEAR_PROBE_LENGTH; i++) {
403 const hashmap_uint32_t index = curr + i;
404
405 if (m->data[index].in_use) {
406 if (m->comparer(m->data[index].key, m->data[index].key_len, key, len)) {
407 /* Blank out the fields including in_use */
408 memset(&m->data[index], 0, sizeof(struct hashmap_element_s));
409
410 /* Reduce the size */
411 m->size--;
412
413 return 0;
414 }
415 }
416 }
417
418 return 1;
419}

421const void *hashmap_remove_and_return_key(struct hashmap_s *const m,
422 const void *const key,
423 const hashmap_uint32_t len) {
424 hashmap_uint32_t i, curr;
425
426 if ((HASHMAP_NULL == key) || (0 == len)) {
427 return HASHMAP_NULL;
428 }
429
430 curr = hashmap_hash_helper_int_helper(m, key, len);
431
432 /* Linear probing, if necessary */
433 for (i = 0; i < HASHMAP_LINEAR_PROBE_LENGTH; i++) {
434 const hashmap_uint32_t index = curr + i;
435
436 if (m->data[index].in_use) {
437 if (m->comparer(m->data[index].key, m->data[index].key_len, key, len)) {
438 const void *const stored_key = m->data[index].key;
439
440 /* Blank out the fields */
441 memset(&m->data[index], 0, sizeof(struct hashmap_element_s));
442
443 /* Reduce the size */
444 m->size--;
445
446 return stored_key;
447 }
448 }
449 }
450
451 return HASHMAP_NULL;
452}

454int hashmap_iterate(const struct hashmap_s *const m,
455 int (*f)(void *const, void *const), void *const context) {
456 hashmap_uint32_t i;
457
458 for (i = 0; i < (hashmap_capacity(m) + HASHMAP_LINEAR_PROBE_LENGTH); i++) {
459 if (m->data[i].in_use) {
460 if (!f(context, m->data[i].data)) {
461 return 1;
462 }
463 }
464 }
465
466 return 0;
467}

469int hashmap_iterate_pairs(struct hashmap_s *const m,
470 int (*f)(void *const,
471 struct hashmap_element_s *const),
472 void *const context) {
473 hashmap_uint32_t i;
474 struct hashmap_element_s *p;
475 int r;
476
477 for (i = 0; i < (hashmap_capacity(m) + HASHMAP_LINEAR_PROBE_LENGTH); i++) {
478 p = &m->data[i];
479 if (p->in_use) {
480 r = f(context, p);
481 switch (r) {
482 case -1: /* remove item */
483 memset(p, 0, sizeof(struct hashmap_element_s));
484 m->size--;
485 break;
486 case 0: /* continue iterating */
487 break;
488 default: /* early exit */
489 return 1;
490 }
491 }
492 }
493 return 0;
494}

623HASHMAP_ALWAYS_INLINE int
624hashmap_hash_helper(const struct hashmap_s *const m, const void *const key,
625 const hashmap_uint32_t len,
626 hashmap_uint32_t *const out_index) {
627 hashmap_uint32_t curr;
628 hashmap_uint32_t i;
629 hashmap_uint32_t first_free;
630
631 /* If full, return immediately */
632 if (hashmap_num_entries(m) == hashmap_capacity(m)) {
633 return 0;
634 }
635
636 /* Find the best index */
637 curr = hashmap_hash_helper_int_helper(m, key, len);
638 first_free = ~0u;
639
640 for (i = 0; i < HASHMAP_LINEAR_PROBE_LENGTH; i++) {
641 const hashmap_uint32_t index = curr + i;
642
643 if (!m->data[index].in_use) {
644 first_free = (first_free < index) ? first_free : index;
645 } else if (m->comparer(m->data[index].key, m->data[index].key_len, key,
646 len)) {
647 *out_index = index;
648 return 1;
649 }
650 }
651
652 // Couldn't find a free element in the linear probe.
653 if (~0u == first_free) {
654 return 0;
655 }
656
657 *out_index = first_free;
658 return 1;
659}

674/*
675 * Doubles the size of the hashmap, and rehashes all the elements
676 */
677HASHMAP_ALWAYS_INLINE int hashmap_rehash_helper(struct hashmap_s *const m) {
678 struct hashmap_create_options_s options;
679 struct hashmap_s new_m;
680 int flag;
681
682 memset(&options, 0, sizeof(options));
683 options.initial_capacity = hashmap_capacity(m) * 2;
684 options.hasher = m->hasher;
685
686 if (0 == options.initial_capacity) {
687 return 1;
688 }
689
690 flag = hashmap_create_ex(options, &new_m);
691
692 if (0 != flag) {
693 return flag;
694 }
695
696 /* copy the old elements to the new table */
697 flag = hashmap_iterate_pairs(m, hashmap_rehash_iterator,
698 HASHMAP_PTR_CAST(void *, &new_m));
699
700 if (0 != flag) {
701 return flag;
702 }
703
704 hashmap_destroy(m);
705
706 /* put new hash into old hash structure by copying */
707 memcpy(m, &new_m, sizeof(struct hashmap_s));
708
709 return 0;
710}
