54radial_window()
55{
56 position = real3((globalVertexIdx.x - AC_nmin.x) * AC_ds.x,
57 (globalVertexIdx.y - AC_nmin.y) * AC_ds.y,
58 (globalVertexIdx.z - AC_nmin.z) * AC_ds.z)
59 center = real3(((AC_ngrid.x-1) * AC_ds.x)/2.0,
60 ((AC_ngrid.y-1) * AC_ds.y)/2.0,
61 ((AC_ngrid.z-1) * AC_ds.z)/2.0)
62 displacement = position-center;
63 distance_to_center = sqrt(AC_dot(displacement,displacement))
64 return distance_to_center < AC_window_radius ? 1.0 : 0.0;
65}
66gaussian_window()
67{
68 position = real3((globalVertexIdx.x - AC_nmin.x) * AC_ds.x,
69 (globalVertexIdx.y - AC_nmin.y) * AC_ds.y,
70 (globalVertexIdx.z - AC_nmin.z) * AC_ds.z)
71 center = real3(((AC_ngrid.x-1) * AC_ds.x)/2.0,
72 ((AC_ngrid.y-1) * AC_ds.y)/2.0,
73 ((AC_ngrid.z-1) * AC_ds.z)/2.0)
74 displacement = position-center;
75 distance_to_center = sqrt(AC_dot(displacement,displacement))
76 return exp(-(distance_to_center / AC_window_radius) * (distance_to_center / AC_window_radius));
77}
