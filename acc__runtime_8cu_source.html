195static bool
196is_valid_configuration(const Volume dims, const Volume tpb, const AcKernel kernel)
197{
198 if (!tpb_is_legal(tpb)) return false;
199 if (dims.x == 1 && dims.y == 1) return true;
200 if (dims.x == 1 && dims.z == 1) return true;
201 if (dims.y == 1 && dims.z == 1) return true;
202 const auto prop = get_device_prop();
203 const size_t warp_size = prop.warpSize;
204 const size_t xmax = (size_t)(warp_size * ceil_div(dims.x,warp_size));
205 const size_t ymax = (size_t)(warp_size * ceil_div(dims.y,warp_size));
206 const size_t zmax = (size_t)(warp_size * ceil_div(dims.z,warp_size));
207 const bool too_large = (tpb.x > xmax) || (tpb.y > ymax) || (tpb.z > zmax);
208 if (is_coop_raytracing_kernel(kernel))
209 {
210 int maxBlocksPerSM{};
211 ERRCHK_CUDA_ALWAYS(
212 acOccupancyMaxActiveBlocksPerMultiprocessor(
213 &maxBlocksPerSM,
214 (void*)kernels[kernel],
215 tpb.x*tpb.y*tpb.z,
216 0
217 ));
218 const auto bpg = get_bpg(dims,to_volume(tpb));
219 const int totalMaxBlocks = get_device_prop().multiProcessorCount*maxBlocksPerSM;
220 if ((int)(bpg.x*bpg.y*bpg.z) > totalMaxBlocks) return false;
221 }
222 if (raytracing_step_direction(kernel).x)
223 {
224 if (((int)tpb.y - (x_ray_shared_mem_block_size)) != 0) return false;
225 }
226
227 else
228 {
229 //TP: in most cases this a reasonable limitation but at least theoretically the shape of the threadblock might be more important
230 // than warp considerations. So impose this limitation only if the user allows it
231 if (sparse_autotuning && (dims.x >= warp_size && tpb.x % warp_size != 0)) return false;
232 }
233
234 if (kernel_reduces_profile(kernel))
235 {
236 if (tpb.y > (size_t)max_tpb_for_reduce_kernels.y) return false;
237 if (tpb.z > (size_t)max_tpb_for_reduce_kernels.z) return false;
238 //TP: if we enforce that tpb.x is a multiple of the warp size then
239 //can easily do warp reduce while doing a reduction whose result is not x-dependent --> major saving in memory and performance increase
240 if (dims.x >= warp_size && tpb.x % warp_size != 0) return false;
241 }
242// const bool single_tb = (tpb.x >= dims.x) && (tpb.y >= dims.y) && (tpb.z >= dims.z);
243
244 //TP: if not utilizing the whole warp invalid, expect if dims are so small that could not utilize a whole warp
245 const bool not_full_warp = (tpb.x*tpb.y*tpb.z < warp_size);
246 if (not_full_warp && is_large_launch(dims)) return false;
247
248 switch (IMPLEMENTATION) {
249 case IMPLICIT_CACHING: {
250
251 if (too_large)
252 return false;
253
254 return true;
255 }
256 case EXPLICIT_CACHING_4D_BLOCKING: // Fallthrough
257 if (tpb.z > 1) return false;
258 [[fallthrough]];
259 case EXPLICIT_CACHING: // Fallthrough
260 case EXPLICIT_CACHING_3D_BLOCKING: {
261
262 // For some reason does not work without this
263 // Probably because of break vs continue when fetching (some threads
264 // quit too early if the dims are not divisible)
265 return !(dims.x % tpb.x) && !(dims.y % tpb.y) && !(dims.z % tpb.z);
266 }
267 case EXPLICIT_PINGPONG_txw: {
268 return (tpb.y == 1) && (tpb.z == 1);
269 }
270 case EXPLICIT_PINGPONG_txy: {
271 return (tpb.z == 1);
272 }
273 case EXPLICIT_PINGPONG_txyblocked: {
274 return (tpb.z == 1);
275 }
276 case EXPLICIT_PINGPONG_txyz: {
277 return true;
278 }
279 case EXPLICIT_ROLLING_PINGPONG: {
280 // OK for every other rolling pingpong implementation
281 // return true;
282
283 // Required only when unrolling smem loads
284 // Ensures two unrolls is enough to fill the smem buffer
285 return (2 * tpb.x >= STENCIL_WIDTH - 1 + tpb.x) &&
286 (2 * tpb.y >= STENCIL_HEIGHT - 1 + tpb.y);
287 }
288 default: {
289 ERROR("Invalid IMPLEMENTATION in is_valid_configuration");
290 return false;
291 }
292 }

295const bool SHARED_MEM_Z_RAYS = false;
296size_t
297get_smem(const AcKernel kernel, const Volume tpb, const size_t stencil_order,
298 const size_t bytes_per_elem)
299{
300 if (is_raytracing_kernel(kernel) && raytracing_step_direction(kernel).x)
301 {
302 //TP: we pad the y dimension by one to avoid bank conflicts
303 return bytes_per_elem*(tpb.y+1)*tpb.z*(x_ray_shared_mem_block_size+2)*num_fields_ray_accessed_read_and_written(kernel);
304 }
305 if (is_raytracing_kernel(kernel) && raytracing_step_direction(kernel).z && SHARED_MEM_Z_RAYS)
306 {
307 return bytes_per_elem*(tpb.x+2)*(tpb.y+2)*(z_ray_shared_mem_block_size+2)*num_fields_ray_accessed_read_and_written(kernel);
308 }
309 switch (IMPLEMENTATION) {
310 case IMPLICIT_CACHING: {
311 return 0;
312 }
313 case EXPLICIT_CACHING: {
314 return (tpb.x + stencil_order) * (tpb.y + stencil_order) * tpb.z *
315 bytes_per_elem;
316 }
317 case EXPLICIT_CACHING_3D_BLOCKING: {
318 return (tpb.x + stencil_order) * (tpb.y + stencil_order) *
319 (tpb.z + stencil_order) * bytes_per_elem;
320 }
321 case EXPLICIT_CACHING_4D_BLOCKING: {
322 return (tpb.x + stencil_order) * (tpb.y + stencil_order) * tpb.z *
323 (NUM_FIELDS)*bytes_per_elem;
324 }
325 case EXPLICIT_PINGPONG_txw: {
326 return 2 * (tpb.x + stencil_order) * NUM_FIELDS * bytes_per_elem;
327 }
328 case EXPLICIT_PINGPONG_txy: {
329 return 2 * (tpb.x + stencil_order) * (tpb.y + stencil_order) *
330 bytes_per_elem;
331 }
332 case EXPLICIT_PINGPONG_txyblocked: {
333 const size_t block_size = 7;
334 return 2 * (tpb.x + stencil_order) * (tpb.y + stencil_order) * block_size *
335 bytes_per_elem;
336 }
337 case EXPLICIT_PINGPONG_txyz: {
338 return 2 * (tpb.x + stencil_order) * (tpb.y + stencil_order) *
339 (tpb.z + stencil_order) * bytes_per_elem;
340 }
341 case EXPLICIT_ROLLING_PINGPONG: {
342 // tpbxy slices with halos
343 // tpbz depth + 1 rolling cache slab
344 return EXPLICIT_ROLLING_PINGPONG_BLOCKSIZE * (tpb.x + stencil_order) *
345 (tpb.y + stencil_order) * (tpb.z + 1) * bytes_per_elem;
346 }
347 default: {
348 ERROR("Invalid IMPLEMENTATION in get_smem");
349 return (size_t)-1;
350 }
351 }

518static void
519acReadOptimTBConfigs(const Volume block_factors_volume)
520{
521 const int3 block_factors = to_int3(block_factors_volume);
522 if (!file_exists(autotune_csv_path)) return;
523 const char* filename = autotune_csv_path;
524 FILE *file = fopen ( filename, "r" );
525 string_vec entries[1000];
526 memset(entries,0,sizeof(string_vec)*1000);
527 const int n_entries = get_csv_entries(entries,file);
528
529 for (int i = 0; i < n_entries; ++i)
530 {
531 string_vec entry = entries[i];
532 if (entry.size == 17)
533 {
534 int kernel_index = atoi(entry.data[1]);
535 int3 read_dims = {atoi(entry.data[2]), atoi(entry.data[3]), atoi(entry.data[4])};
536 int3 tpb = {atoi(entry.data[5]), atoi(entry.data[6]), atoi(entry.data[7])};
537 [[maybe_unused]] double time = atof(entry.data[8]);
538 int3 read_block_factors = {atoi(entry.data[10]), atoi(entry.data[11]), atoi(entry.data[12])};
539 int3 read_raytracing_factors = {atoi(entry.data[13]), atoi(entry.data[14]), atoi(entry.data[15])};
540 int was_sparse = atoi(entry.data[16]);
541 if (read_block_factors == block_factors && read_raytracing_factors == raytracing_subblock && was_sparse == sparse_autotuning)
542 {
543 TBConfig c = (TBConfig){AcKernel(kernel_index),read_dims,(dim3){(uint32_t)tpb.x, (uint32_t)tpb.y, (uint32_t)tpb.z}};
544 tbconfigs.push_back(c);
545 }
546 }
547 for (size_t elem = 0; elem < entry.size; ++elem)
548 free((char*)entry.data[elem]);
549 free_str_vec(&entry);
550 }
551 fclose(file);
552}
