185static HOST_DEVICE_INLINE AcComplex
186sqrt(const AcComplex& a)
187{
188 const AcReal r = sqrt(a.x*a.x + a.y*a.y);
189 const AcReal y_sign = (a.y < AcReal(0.0)) ? AcReal(-1.0) : AcReal(1.0);
190 return (AcComplex){sqrt((r + a.x)*AcReal(0.5)),y_sign * sqrt((r-a.x)*AcReal(0.5))};
191}

695static HOST_DEVICE_INLINE AcReal
696multm2_sym(const AcMatrix m)
697{
698 AcReal res = m.data[0][0]*m.data[0][0];
699 for(int i = 1; i < 3; ++i)
700 {
701 res += m.data[i][i]*m.data[i][i];
702 for(int j = 0; j < i; ++j)
703 {
704 res += 2*m.data[i][j]*m.data[i][j];
705 }
706 }
707 return res;
708}
