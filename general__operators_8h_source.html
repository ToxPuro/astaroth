9u_dot_grad(Field3 f, Matrix m,real3 u){
10 suppress_unused_warning(f)
11 real3 res = m*u
12 if(AC_coordinate_system == AC_SPHERICAL_COORDINATES)
13 {
14 res.x = res.x - AC_INV_R*(u.y*f.y+u.z*f.z)
15 res.y = res.y + AC_INV_R*(u.y*f.x-u.z*f.z*AC_COT)
16 res.z = res.z + AC_INV_R*(u.z*f.x+u.z*f.y*AC_COT)
17 }
18 else if(AC_coordinate_system == AC_CYLINDRICAL_COORDINATES)
19 {
20 res.x = res.x - AC_INV_CYL_R*(u.y*f.y)
21 res.y = res.y + AC_INV_CYL_R*(u.y*f.x)
22 }
23 return res;
24}

411bij(Field3 v)
412{
413 d2A = get_d2A(v)
414
415 Matrix res;
416 res[0][0] = d2A[1][0][2] - d2A[2][0][1]
417 res[0][1] = d2A[1][1][2] - d2A[2][1][1]
418 res[0][2] = d2A[1][2][2] - d2A[2][2][1]
419
420 res[1][0] = d2A[2][0][0] - d2A[0][0][2]
421 res[1][1] = d2A[2][1][0] - d2A[0][1][2]
422 res[1][2] = d2A[2][2][0] - d2A[0][2][2]
423
424 res[2][0] = d2A[0][0][1] - d2A[1][0][0]
425 res[2][1] = d2A[0][1][1] - d2A[1][1][0]
426 res[2][2] = d2A[0][2][1] - d2A[1][2][0]
427
428 if(AC_coordinate_system == AC_SPHERICAL_COORDINATES)
429 {
430 res[3-1][2-1]=res[3-1][2-1]+dery(v.y)*AC_INV_R
431 res[2-1][3-1]=res[2-1][3-1]-derz(v.z)*AC_INV_R
432 res[1-1][3-1]=res[1-1][3-1]+derz(v.z)*AC_INV_R*AC_COT
433 res[3-1][1-1]=res[3-1][1-1]+derx(v.y)*AC_INV_R-v.y*(AC_INV_R*AC_INV_R)
434 res[2-1][1-1]=res[2-1][1-1]-derx(v.z)*AC_INV_R+v.z*(AC_INV_R*AC_INV_R)
435 res[1-1][2-1]=res[1-1][2-1]+dery(v.z)*AC_INV_R*AC_COT-v.z*(AC_INV_R*AC_INV_R)*(AC_INV_SIN_THETA*AC_INV_SIN_THETA)
436 }
437 else if(AC_coordinate_system == AC_CYLINDRICAL_COORDINATES)
438 {
439 res[3-1][2-1]=res[3-1][2-1]+ dery(v.y)*AC_INV_CYL_R
440 res[3-1][1-1]=res[3-1][1-1]+ derx(v.y)*AC_INV_CYL_R-v.y*(AC_INV_CYL_R*AC_INV_CYL_R)
441 }
442
443
444 return res;
445}
