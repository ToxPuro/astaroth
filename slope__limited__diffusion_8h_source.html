653get_x_fluxes(Field f,Field characteristic_speed, real fdiff_limit, real h_slope_limited, real nlf, bool ln_field)
654{
655 if(!AC_dimension_inactive.x)
656 {
657 cs = get_x_interpolated_characteristic_speeds(characteristic_speed)
658 interface_values = get_x_interface_values(f,ln_field)
659
660 left_diff = ln_field ? sld_diff_left_exp(f) : sld_diff_left(f)
661 left_add = ln_field ? sld_add_left_exp(f) : sld_add_left(f)
662 left_interface_diff = interface_values.left - interface_values.left_left
663
664 int tmp = (((left_interface_diff)*left_diff) > 0.0)
665 left_slope_ratio = tmp*(left_interface_diff)/left_diff
666 tmp = tmp*(abs(left_add/left_diff) > fdiff_limit)
667 left_diff = (1-tmp)*left_diff + tmp*sign(left_add,left_diff)/fdiff_limit
668
669 left_Q = pow(min(1.0,h_slope_limited*left_slope_ratio),nlf)
670 left_flux = 0.5*cs.left*left_Q*(left_interface_diff)
671
672
673 right_diff = ln_field ? sld_diff_right_exp(f) : sld_diff_right(f)
674 right_add = ln_field ? sld_add_right_exp(f) : sld_add_right(f)
675 right_interface_diff = interface_values.right_right - interface_values.right
676
677 tmp = (((right_interface_diff)*right_diff) > 0.0)
678 right_slope_ratio = tmp*(right_interface_diff)/right_diff
679 tmp = tmp*(abs(right_add/right_diff) > fdiff_limit)
680 right_diff = (1-tmp)*right_diff + tmp*sign(right_add,right_diff)/fdiff_limit
681
682 right_Q = pow(min(1.0,h_slope_limited*right_slope_ratio),nlf)
683 right_flux = 0.5*cs.right*right_Q*(right_interface_diff)
684 return sld_flux(left_flux,right_flux)
685 }
686 return sld_flux(0.0,0.0)

790get_slope_limited_divergence(sld_fluxes fluxes)
791{
792 real divx
793 if (AC_coordinate_system == AC_SPHERICAL_COORDINATES)
794 {
795 divx = (AC_x12[vertexIdx.x]*AC_x12[vertexIdx.x]*fluxes.x.right - AC_x12[vertexIdx.x-1]*AC_x12[vertexIdx.x-1]*fluxes.x.left)*AC_INV_R*AC_INV_R
796 }
797 else if (AC_coordinate_system == AC_CYLINDRICAL_COORDINATES)
798 {
799 divx = (AC_x12[vertexIdx.x]*fluxes.x.right - AC_x12[vertexIdx.x-1]*fluxes.x.left)*AC_INV_R
800 }
801 else
802 {
803 divx = fluxes.x.right - fluxes.x.left
804 }
805
806 if(!AC_nonequidistant_grid.x)
807 {
808 divx *= AC_inv_ds.x
809 }
810 else
811 {
812 dx12 = AC_x12[vertexIdx.x] - AC_x12[vertexIdx.x-1]
813 divx /= dx12
814 }
815
816 real divy
817 if (AC_coordinate_system == AC_SPHERICAL_COORDINATES)
818 {
819 divy = (AC_sinth12[vertexIdx.y]*fluxes.y.right - AC_sinth12[vertexIdx.y-1]*fluxes.y.left)
820 *AC_INV_SIN_THETA
821 } else
822 {
823 divy = fluxes.y.right - fluxes.y.left
824 }
825 if (AC_coordinate_system == AC_SPHERICAL_COORDINATES || AC_coordinate_system == AC_CYLINDRICAL_COORDINATES)
826 {
827 divy *= AC_INV_R
828 }
829 if(!AC_nonequidistant_grid.y)
830 {
831 divy *= AC_inv_ds.y
832 }
833 else
834 {
835 dy12 = AC_y12[vertexIdx.y] - AC_y12[vertexIdx.y-1]
836 divy /= dy12
