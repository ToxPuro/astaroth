56static inline ASTNode*
57astnode_create(const NodeType type, ASTNode* lhs, ASTNode* rhs)
58{
59 ASTNode* node = calloc(1, sizeof(node[0]));
60
61 static int id_counter = 0;
62 node->id = id_counter++;
63 node->type = type;
64 node->lhs = lhs;
65 node->rhs = rhs;
66 node->buffer = NULL;
67
68 node->token = 0;
69 node->prefix = node->infix = node->postfix = 0;
70
71 if (lhs)

73
74 if (rhs)
